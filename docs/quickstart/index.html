Quickstart

Move from typed plans to audit-ready traces in a two-week compliance pilot.

1. Emit PlanLang

import { plan } from "@praeon/sdk";

const typedPlan = plan({
  actor: { type: "agent", id: "claimsbot" },
  capabilities: [{ tool: "jira", name: "update", scopes: ["project:claims"] }],
  constraints: { budget: { currency: "USD", max: 50 }, riskLevel: "low" },
  steps: [{
    id: "s1",
    action: { tool: "jira", capability: "update", params: { issueId: "CLAIM-42" } },
    preconditions: ["issue.status in ['Open','Pending']"]
  }],
  approvals: [{ stepId: "s1", type: "human", role: "claims.manager" }],
  policyRef: ["policy://claims/low-risk"]
});

2. Pre-Flight

import { preflight } from "@praeon/sdk";

await preflight({
  plan: typedPlan,
  environment: "pilot",
  adapters: ["jira"],
});

3. Approve & Execute

import { approve, execute } from "@praeon/sdk";

await approve({ planId: typedPlan.id, approver: "manager.alice" });
const trace = await execute({ planId: typedPlan.id });

4. Export Evidence

import { exportTrace } from "@praeon/sdk";

await exportTrace({
  planId: typedPlan.id,
  format: "audit",
  destination: "s3://praeon-audit/pilot"
});
